# Macro Cockpit Redesign Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Rebuild `macro.html` into a decision-first macro cockpit that leads with a clear market judgement, keeps the upcoming event schedule visible, and moves raw explanations/news into drill-down surfaces.

**Architecture:** Keep the existing static GitHub Pages architecture and current `data/macro-latest.json` contract. Add a front-end view-model layer inside `macro.html` that derives `headlineDecision`, evidence cards, support/counter evidence, watchlist, and asset temperature from existing JSON. Replace the current long two-column page with a balanced cockpit layout and a lower risk schedule.

**Tech Stack:** Static HTML/CSS/vanilla JavaScript, existing JSON generated by `scripts/macro-akshare.py`, Node.js for syntax/smoke checks.

---

## File Structure

- Modify `macro.html`
  - Owns layout, derived view model, rendering, drawer/modal behavior, and page copy.
  - This is intentionally still a single file because the repo currently ships static pages without a bundler.
- Create `tests/macro-cockpit-smoke.js`
  - Performs dependency-free checks on `macro.html`: script syntax, required DOM anchors, and forbidden user-hostile copy.
- No changes to `scripts/macro-akshare.py` in this pass.
  - The first implementation derives cockpit structures on the client. Only move this logic into Python later if the UI needs richer precomputed judgement text.

```js
function numberFrom(value) {
  const match = String(value || "").replaceAll(",", "").match(/-?\d+(\.\d+)?/);
  return match ? Number(match[0]) : null;
}

function metricByName(items, needle) {
  return (items || []).find((item) => String(item.name || "").includes(needle));
}

function eventById(id) {
  return state.events.find((event) => event.id === id);
}

function buildHeadlineDecision() {
  const realRate = metricByName(state.rateMetrics, "实际利率");
  const dxy = metricByName(state.prices, "美元指数");
  const vix = metricByName(state.prices, "VIX");
  const fng = metricByName(state.cryptoMetrics, "恐慌贪婪");
  const goldOil = metricByName(state.ratios, "金油比");
  const ethBtc = metricByName(state.ratios, "ETH/BTC");

  const defensiveVotes = [];
  const riskVotes = [];

  if (numberFrom(realRate?.value) >= 2) defensiveVotes.push("实际利率仍高，黄金和风险资产都承压");
  if (numberFrom(dxy?.value) >= 98) defensiveVotes.push("美元指数不弱，全球流动性偏紧");
  if ((goldOil?.changePct || 0) > 5) defensiveVotes.push("金油比上行，增长担忧或避险需求抬头");
  if ((ethBtc?.changePct || 0) < 0) defensiveVotes.push("ETH/BTC 走弱，币圈内部风险偏好不足");

  if (numberFrom(vix?.value) < 18) riskVotes.push("VIX 不高，市场还不是系统性恐慌");
  if (numberFrom(fng?.value) > 25 && numberFrom(fng?.value) < 75) riskVotes.push("币圈情绪未到极端，不适合硬做反转");

  let headline = "偏观望，等待关键数据确认";
  if (defensiveVotes.length >= 3 && riskVotes.length) headline = "偏防守，但不是极端避险";
  else if (defensiveVotes.length >= 3) headline = "偏防守，优先控制风险";
  else if (riskVotes.length >= 2 && defensiveVotes.length <= 1) headline = "偏中性，风险偏好尚未恶化";

  return {
    headline,
    reason: [...defensiveVotes.slice(0, 2), ...riskVotes.slice(0, 1)].join("；") || "核心指标分歧较大，今天先看关键数据和新闻是否给出方向。",
    actions: [
      { label: "黄金", value: defensiveVotes.length >= 2 ? "观察偏多" : "等实际利率" },
      { label: "BTC", value: (ethBtc?.changePct || 0) < 0 ? "不追" : "观察资金流" },
      { label: "美股", value: numberFrom(vix?.value) < 18 ? "等数据" : "控回撤" },
      { label: "现金", value: "保留机动" },
    ],
    support: defensiveVotes,
    counter: riskVotes,
  };
}

function buildEvidenceCards(decision) {
  const realRate = metricByName(state.rateMetrics, "实际利率");
  const dxy = metricByName(state.prices, "美元指数");
  const goldOil = metricByName(state.ratios, "金油比");
  const ethBtc = metricByName(state.ratios, "ETH/BTC");
  const fng = metricByName(state.cryptoMetrics, "恐慌贪婪");
  const vix = metricByName(state.prices, "VIX");

  return [
    {
      title: "潮水",
      verdict: numberFrom(realRate?.value) >= 2 || numberFrom(dxy?.value) >= 98 ? "偏紧" : "中性",
      note: `${realRate?.value || "实际利率缺失"}；${dxy?.value ? `DXY ${dxy.value}` : "DXY 缺失"}`,
    },
    {
      title: "避险",
      verdict: (goldOil?.changePct || 0) > 5 ? "升温" : "不极端",
      note: goldOil ? `金油比 ${goldOil.value}，近月 ${formatPct(goldOil.changePct)}` : "等待金油比",
    },
    {
      title: "币圈",
      verdict: (ethBtc?.changePct || 0) < 0 ? "偏防守" : "观察扩散",
      note: ethBtc ? `ETH/BTC ${ethBtc.value}，近月 ${formatPct(ethBtc.changePct)}` : "等待 ETH/BTC",
    },
    {
      title: "情绪",
      verdict: "未极端",
      note: `${vix?.value ? `VIX ${vix.value}` : "VIX 缺失"}；${fng?.value ? `恐慌贪婪 ${fng.value}` : "情绪缺失"}`,
    },
  ];
}

function buildWatchlist(items) {
  const scored = items
    .map((item) => {
      const title = item.title || "";
      let score = 1;
      let why = "只在能解释资产变化时关注";
      if (/PCE|CPI|非农|FOMC|美联储|初请|失业率/.test(title)) {
        score = 3;
        why = "可能改变降息预期和美元/美债方向";
      } else if (/OPEC|原油|中东|伊朗|以色列|俄乌|关税|特朗普/.test(title)) {
        score = 2;
        why = "可能影响通胀、原油和黄金避险需求";
      } else if (/BTC|比特币|ETF|稳定币|ETH/.test(title)) {
        score = 2;
        why = "可能影响币圈资金流和风险偏好";
      }
      return { ...item, score, why };
    })
    .sort((a, b) => b.score - a.score || a.date - b.date);

  return scored.slice(0, 5);
}

function sevenDayItems() {
  const now = new Date();
  const end = new Date(now);
  end.setDate(now.getDate() + 7);
  return calendarItems()
    .filter(itemMatches)
    .filter((item) => item.date >= new Date(now.getFullYear(), now.getMonth(), now.getDate() - 2) && item.date <= end)
    .sort((a, b) => a.date - b.date);
}

function formatPct(value) {
  if (typeof value !== "number" || Number.isNaN(value)) return "-";
  return `${value >= 0 ? "+" : ""}${value.toFixed(1)}%`;
}
```

```js
function renderHeadline() {
  const decision = buildHeadlineDecision();
  els.headlineDecision.textContent = decision.headline;
  els.headlineReason.textContent = decision.reason;
  els.assetActions.innerHTML = decision.actions
    .map((item) => `<span class="action-chip"><b>${escapeHtml(item.label)}</b>：${escapeHtml(item.value)}</span>`)
    .join("");
  els.dataHealth.textContent = state.sources.some((source) => source.status === "failed")
    ? "部分数据源失败，核心报告可用"
    : "核心数据源正常";
  return decision;
}

function renderEvidenceCards(decision) {
  els.evidenceCards.innerHTML = buildEvidenceCards(decision)
    .map((card) => `
      <article class="evidence-card">
        <b>${escapeHtml(card.title)}：${escapeHtml(card.verdict)}</b>
        <p>${escapeHtml(card.note)}</p>
      </article>
    `)
    .join("");

  els.supportEvidence.innerHTML = (decision.support.length ? decision.support : ["没有足够强的单边证据，先等待关键数据。"])
    .slice(0, 4)
    .map((item) => `<div class="evidence-note"><span>${escapeHtml(item)}</span></div>`)
    .join("");

  els.counterEvidence.innerHTML = (decision.counter.length ? decision.counter : ["暂时缺少明显反向证据，但仍需看新闻和数据是否突然转向。"])
    .slice(0, 4)
    .map((item) => `<div class="evidence-note"><span>${escapeHtml(item)}</span></div>`)
    .join("");
}

function renderWatchlist() {
  const items = buildWatchlist(sevenDayItems());
  els.watchlist.innerHTML = items.length
    ? items.map((item) => `
        <button class="watch-card" data-id="${escapeHtml(item.id)}" type="button">
          <b>${item.score === 3 ? "高" : item.score === 2 ? "中" : "低"}｜${escapeHtml(item.title)}</b>
          <p>${escapeHtml(item.why)}</p>
        </button>
      `).join("")
    : `<div class="watch-card"><b>暂无必须盯事件</b><p>当前 7 天窗口内没有足以改变总判断的事件。</p></div>`;
}

function renderRiskSchedule() {
  const items = buildWatchlist(sevenDayItems()).concat(sevenDayItems().filter((item) => !buildWatchlist(sevenDayItems()).some((watch) => watch.id === item.id))).slice(0, 12);
  els.riskSchedule.innerHTML = items.length
    ? items.map((item) => `
        <button class="schedule-item" data-id="${escapeHtml(item.id)}" type="button">
          <b>${escapeHtml(item.date.toLocaleDateString("zh-CN", { month: "2-digit", day: "2-digit" }))}</b>
          <p>${escapeHtml(item.title)}</p>
          <span class="pill">${escapeHtml(item.label)}</span>
        </button>
      `).join("")
    : `<div class="schedule-item"><b>-</b><p>当前筛选下没有风险事件。</p><span class="pill">空</span></div>`;
}
```
